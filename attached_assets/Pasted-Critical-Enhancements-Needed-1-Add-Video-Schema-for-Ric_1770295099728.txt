10. SEO Middleware: Add Caching Layer
Current middleware re-reads HTML on every request. Add:
typescriptconst metaCache = new Map<string, string>();

export function createSeoMiddleware(staticPath: string) {
  const baseHtml = readFileSync(path.join(staticPath, "index.html"), "utf-8");
  
  return (req, res, next) => {
    const cacheKey = req.path;
    
    if (metaCache.has(cacheKey)) {
      return res.send(metaCache.get(cacheKey));
    }
    
    const meta = getPageMeta(req.path);
    const modifiedHtml = injectMeta(baseHtml, meta);
    
    metaCache.set(cacheKey, modifiedHtml); // Cache for 5min
    setTimeout(() => metaCache.delete(cacheKey), 300000);
    
    res.send(modifiedHtml);
  };
}
